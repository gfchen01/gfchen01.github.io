462  if (leaf->key_field.at(i) == key){

464  leaf->value_field.erase(leaf->value_field.begin() + i);
465  break;
466  }
467  }
468 
469  if (leaf == root) return true;
470  // Violate capacity constraint for leaf and non-root node
471  else if (leaf->key_field.size() < degree / 2){ // (int)degree/2 is equivalent to (degree - 1)/2's upper integer
472  BP_NODE_T* sibling;
473  BP_NODE_T* parent = leaf->parent;
474  key_t sep_key_in_parent;
475  size_t i;
476  bool is_predecessor;
477  for (i = 0; i < parent->ptr_field.size(); ++i){
478  if (parent->ptr_field.at(i) == leaf) break;
479  }
480  assert(i < parent->ptr_field.size());
481  if (i < parent->ptr_field.size() - 1){ // Not the last element
482  sibling = parent->ptr_field.at(i + 1);
483  is_predecessor = true;
484  sep_key_in_parent = parent->key_field.at(i);
485  }
486  else {
487  sibling = parent->ptr_field.at(i - 1);
488  is_predecessor = false;
489  sep_key_in_parent = parent->key_field.at(i - 1);
490  }

495  // Exchange the variables for convenience here
